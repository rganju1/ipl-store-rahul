<script>

const teams = {

  mi: [

    { name: "Rohit Sharma", number: 45 },

    { name: "Jasprit Bumrah", number: 93 }

  ],

  csk: [

    { name: "MS Dhoni", number: 7 },

    { name: "Ruturaj Gaikwad", number: 31 }

  ],

  rcb: [

    { name: "Virat Kohli", number: 18 },

    { name: "Glenn Maxwell", number: 32 }

  ]

};


function updatePlayers() {

  const team = document.getElementById("team").value;

  const playerSelect = document.getElementById("player");


  playerSelect.innerHTML = '<option>Select Player</option>';


  if (teams[team]) {

    teams[team].forEach(p => {

      const opt = document.createElement("option");

      opt.value = JSON.stringify(p);

      opt.text = p.name;

      playerSelect.appendChild(opt);

    });

  }

}


function fillPlayer() {

  const val = document.getElementById("player").value;

  if (!val) return;


  const player = JSON.parse(val);

  document.getElementById("name").value = player.name;

  document.getElementById("number").value = player.number;

  updatePreview();

}


document.getElementById("name").addEventListener("input", updatePreview);

document.getElementById("number").addEventListener("input", updatePreview);


function updatePreview() {

  const name = document.getElementById("name").value;

  const number = document.getElementById("number").value;


  document.getElementById("preview").innerHTML =

    `Preview: ${name} - #${number}`;

}

</script>
